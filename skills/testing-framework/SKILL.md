## Test Types

### 1. Unit Tests
```javascript
// Example: pricing-optimizer test
describe('Pricing Optimizer', () => {
  test('calculates margin correctly', () => {
    const result = calculateMargin(15999, 8500);
    expect(result).toBe(46.8);
  });
  
  test('never recommends below min margin', () => {
    const recommendation = getRecommendation({
      currentPrice: 15999,
      minMargin: 15,
      cogs: 14000
    });
    expect(recommendation.margin).toBeGreaterThanOrEqual(15);
  });
});
```

### 2. Integration Tests
```javascript
// Test skill integration
describe('Competitive Intel Integration', () => {
  test('reads brand context correctly', async () => {
    const context = await readBrandContext();
    expect(context.brandName).toBeDefined();
    expect(context.competitors).toBeInstanceOf(Array);
  });
  
  test('sends WhatsApp alert on price drop', async () => {
    const alert = await sendPriceAlert({
      product: 'test-product',
      priceDrop: 15
    });
    expect(alert.sent).toBe(true);
  });
});
```

### 3. End-to-End Tests

### Pre-Commit Checks
```bash
#!/bin/bash
# .github/hooks/pre-commit

echo "🔍 Running pre-commit checks..."

# Lint Markdown
npx markdownlint skills/

# Validate YAML frontmatter
node scripts/validate-frontmatter.js

# Check skill completeness
node scripts/check-skill-completeness.js

# Run unit tests
npm test

echo "✅ All checks passed!"
```

### Skill Validation Rules
Every skill must have:
- [ ] Valid YAML frontmatter
- [ ] Required fields: name, description, version
- [ ] Dependencies declared
- [ ] Usage examples
- [ ] Safety guardrails documented
- [ ] Credits/attribution

## GitHub Actions Workflow

```yaml
name: Test & Validate

on: [push, pull_request]

jobs:
  test:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v3
      
      - name: Setup Node
        uses: actions/setup-node@v3
        with:
          node-version: '18'
      
      - name: Install dependencies
        run: npm ci
      
      - name: Run linter
        run: npm run lint
      
      - name: Run unit tests
        run: npm test
      
      - name: Run integration tests
        run: npm run test:integration
      
      - name: Validate skills
        run: npm run validate
      
      - name: Upload coverage
        uses: codecov/codecov-action@v3
```

## Gemini Builder Integration

```javascript
// scripts/build-with-gemini.js
const { GoogleGenerativeAI } = require('@google/generative-ai');

const genAI = new GoogleGenerativeAI(process.env.GEMINI_API_KEY);

async function buildSkill(skillName) {
  const model = genAI.getGenerativeModel({ model: 'gemini-pro' });
  
  const prompt = `
    Build a marketing skill for: ${skillName}
    
    Requirements:
    - Follow SKILL.md format
    - Include usage examples
    - Add safety guardrails
    - Optimize for Indian mobile market
    - Use async/await
    - Include error handling
    
    Generate the complete skill code.
  `;
  
  const result = await model.generateContent(prompt);
  const code = result.response.text();
  
  // Validate generated code
  const validation = await validateSkill(code);
  
  if (validation.passed) {
    await saveSkill(skillName, code);
    await runTests(skillName);
    return { success: true };
  } else {
    return { success: false, errors: validation.errors };
  }
}
```
